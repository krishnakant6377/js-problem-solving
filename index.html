Q1. Find the Smaller Angle?
Ans. var Minimal_Angle = (h, m) =>{ const minuteAngle = m * 6; const hourAngle = (h * 30) + (m * 0.5); const angle = Math.abs(hourAngle - minuteAngle); return Math.min(angle, 360 - angle); };

Q2. Check whether the year is Leap year or not?
Ans. var Check_Leap = (year) => { if ( year % 4 === 0 && year % 100 !== 0 || year % 400 === 0){ return "Leap Year"; } else{ return "Non Leap Year"; } };

Q3. Perfect Number Check?
Ans. var Perfect_Check = (N) => { var count = 0; for(var i = 0; i < N; i++){ if(N % i === 0){ count = count + i; } } if(count == N) { return "YES"; } else { return "NO"; } };

Q4. Reverse a Number?
Ans. var Reverse_Number = (N) => { N = N + ""; var b = parseFloat((N.split('').reverse().join(''))); return b; };

Q5. Substring Check?
Ans. var Substring_Check = (S1, S2) => { if (S1.includes(S2)){ return "YES"; } else{ return "NO"; } };
